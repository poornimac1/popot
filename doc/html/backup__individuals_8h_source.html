674  {
675  // The update of the velocity is done according to the equation :
676  // v_i(t+1) = w v_i(t) + x'_i(t) - x_i(t)
677  // where x_i is sampled uniformely from the hypersphere (Gi, Ri)
678  // with Gi the center and Ri the radius and
679  // Gi = 1/3 ( x_i + (x_i + c(p_i - x_i)) + x_i + c(l_i - x_i))
680  // = x_i + c (p_i + l_i - 2x_i)/3
681  // Ri = ||G_i - x_i||
682 
683  // We first check if the local best and personal best are identical by computing the norm of the difference
684  // between both positions
685  double diff_li_pi = 0.0;
686  bool li_equals_pi = false;
687  for(int i = 0 ; i < TSuper::_dimension ; ++i)
688  diff_li_pi += (this->getBestPosition()->getPosition(i) - this->getNeighborhood()->getBest()->getPosition(i))
689  *(this->getBestPosition()->getPosition(i) - this->getNeighborhood()->getBest()->getPosition(i));
690  li_equals_pi = (sqrt(diff_li_pi) <= 1e-5);
691 
692  // We first compute the center of gravity
693  if(li_equals_pi)
694  {
695  for(int i = 0 ; i < TSuper::_dimension ; ++i)
696  gi[i] = this->getPosition(i) + PARTICLE_PARAMS::c() * (this->getBestPosition()->getPosition(i) - this->getPosition(i)) / 2.0;
697  }
698  else
699  {

703  }

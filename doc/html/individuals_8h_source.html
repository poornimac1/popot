00482       void updateVelocity_spso2011(PARTICLE &p)
00483       {
00484 
00485           // The update of the velocity is done according to the equation :
00486           // v_i(t+1) = w v_i(t) + x'_i(t) - x_i(t)
00487           // where x'_i is normally sampled from the hypersphere (Gi, Ri)
00488           // with Gi the center and Ri the radius and
00489           // Gi = 1/3 ( x_i + (x_i + c(p_i - x_i)) + (x_i + c(l_i - x_i)))
00490           //    = 1/3 ( x_i +          p1          +            p2     )))
00491           // or
00492           // Gi = 1/2 (x_i + (x_i + c(p_i - x_i))) if p_i == l_i
00493           //    = 1/2 (x_i +          p1         )
00494           // Ri = ||G_i - x_i||
00495 
00496         double * xpi = new double[p.getPosition().size()];
00497         double * p1 = new double[p.getPosition().size()];
00498         double * p2 = new double[p.getPosition().size()];
00499         double * gr = new double[p.getPosition().size()];
00500 
00501 
00502           size_t i;
00503 
00504           // We first check if the local best and personal best are identical by computing the norm of the difference
00505           // between both positions
00506           // This can be done with pointer comparison
00507           // as the neighborhood holds a pointer to the best personal best
00508           bool li_equals_pi = (&(p.getBestPosition()) == p.getNeighborhood().getBest());
00509 
00510           typename PARTICLE::VECTOR_TYPE& pos = p.getPosition();
00511           typename PARTICLE::VECTOR_TYPE& vel = p.getVelocity();
00512 
00513           // First position
00514           // p1 = xi + c * (pi - xi)
00515           // where pi is the personal best position
00516           for(i = 0 ; i < pos.size() ; ++i)
00517             p1[i] = pos[i] + PARAMS::c() *(p.getBestPosition().getPosition()[i] - pos[i]);
00518           
00519           // Second position
00520           // p2 = xi + c * (li - xi)
00521           // where li is the local best position (within the neighborhood)
00522           for(i = 0 ; i < pos.size() ; ++i)
00523             p2[i] = pos[i] + PARAMS::c() *(p.getNeighborhood().getBest()->getPosition()[i] - pos[i]);
00524           
00525           // Compute the gravity center of p1, p2 and xi
00526           if(!li_equals_pi)
00527             {
00528               // We here consider p1, p2 and xi
00529               for(i = 0 ; i < pos.size() ; ++i)
00530                 gr[i] = 1.0/3.0 * (pos[i] + p1[i] + p2[i]);
00531             }
00532           else
00533             {
00534               // We here consider only p1 (or p2, since they are equal) and xi
00535               for(i = 0 ; i < pos.size() ; ++i)
00536                 gr[i] = 1.0/2.0 * (pos[i] + p1[i]);
00537             }
00538 
00539           // And the radius of the hypersphere
00540           double ri = 0.0;
00541           for(i = 0 ; i < pos.size() ; ++i)
00542             ri += (gr[i] - pos[i])*(gr[i] - pos[i]);
00543           ri = sqrt(ri);
00544 
00545           // Compute the auxiliary position x'_i randomly within the hypersphere (gr, ri);
00546           // To uniformely sample from the hypersphere we uniformely sample a direction
00547           double norm = 0.0;
00548           for(i = 0 ; i < pos.size() ; ++i)
00549             {
00550               xpi[i] = popot::math::normal(0.0,1.0);
00551               norm += xpi[i] * xpi[i];
00552             }
00553           norm = sqrt(norm);
00554           
00555           // And then scale by a random radius
00556           double r = popot::math::uniform_random(0.0,1.0);
00557           for(i = 0 ; i < pos.size() ; ++i)
00558             xpi[i] =  gr[i] + r * ri * xpi[i] / norm;
00559 
00560           // And then update the velocity

00562             vel[i] = PARAMS::w() * vel[i] + xpi[i] - pos[i];

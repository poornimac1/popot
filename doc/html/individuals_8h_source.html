642  virtual void updateVelocity(void)
643  {
644  // The update of the velocity is done according to the equation :
645  // v_i(t+1) = w v_i(t) + x'_i(t) - x_i(t)
646  // where x'_i is normally sampled from the hypersphere (Gi, Ri)
647  // with Gi the center and Ri the radius and
648  // Gi = 1/3 ( x_i + (x_i + c(p_i - x_i)) + (x_i + c(l_i - x_i)))
649  // = 1/3 ( x_i + p1 + p2 )))
650  // or
651  // Gi = 1/2 (x_i + (x_i + c(p_i - x_i))) if p_i == l_i
652  // = 1/2 (x_i + p1 )
653  // Ri = ||G_i - x_i||
654 
655  // We first check if the local best and personal best are identical by computing the norm of the difference
656  // between both positions
657  // This can be done with pointer comparison
658  // as the neighborhood holds a pointer to the best personal best
659  bool li_equals_pi = (this->getBestPosition() == this->getNeighborhood()->getBest());
660 
661  // First position
662  // p1 = xi + c * (pi - xi)
663  // where pi is the personal best position
664  for(int i = 0 ; i < TSuper::_dimension ; ++i)
665  p1[i] = this->getPosition(i) + PARTICLE_PARAMS::c() *(this->getBestPosition()->getPosition(i) - this->getPosition(i));
666 
667  // Second position
668  // p2 = xi + c * (li - xi)
669  // where li is the local best position (within the neighborhood)
670  for(int i = 0 ; i < TSuper::_dimension ; ++i)
671  p2[i] = this->getPosition(i) + PARTICLE_PARAMS::c() *(this->getNeighborhood()->getBest()->getPosition(i) - this->getPosition(i));
672 
673  // Compute the gravity center of p1, p2 and xi
674  if(!li_equals_pi)
675  {
676  // We here consider p1, p2 and xi
677  for(int i = 0 ; i < TSuper::_dimension ; ++i)
678  gr[i] = 1.0/3.0 * (this->getPosition(i) + p1[i] + p2[i]);
679  }
680  else
681  {
682  // We here consider only p1 (or p2, since they are equal) and xi
683  for(int i = 0 ; i < TSuper::_dimension ; ++i)
684  gr[i] = 1.0/2.0 * (this->getPosition(i) + p1[i]);
685  }
686 
687  // And the radius of the hypersphere
688  double ri = 0.0;
689  for(int i = 0 ; i < TSuper::_dimension ; ++i)
690  ri += (gr[i] - this->getPosition(i))*(gr[i] - this->getPosition(i));
691  ri = sqrt(ri);
692 
693  // Compute the auxiliary position x'_i randomly within the hypersphere (gr, ri);
694  // To uniformely sample from the hypersphere we uniformely sample a direction
695  double norm = 0.0;
696  for(int i = 0 ; i < TSuper::_dimension ; ++i)
697  {
698  xpi[i] = popot::math::normal(0.0,1.0);
699  norm += xpi[i] * xpi[i];
700  }
701  norm = sqrt(norm);
702 
703  // And then scale by a random radius
704  for(int i = 0 ; i < TSuper::_dimension ; ++i)
705  xpi[i] = gr[i] + popot::math::uniform_random(0.0,1.0) * ri * xpi[i] / norm;
706 
707  // And then update the velocity
708  for(int i = 0 ; i < TSuper::_dimension ; ++i)
709  this->setVelocity(i, PARTICLE_PARAMS::w() * this->getVelocity(i)
710  + xpi[i] - this->getPosition(i));
711  }
